// Guard clauses, flat
public Response processUser(Request req) {
  validate(req);
  User user = req.getUser();
  requireNonNull(user, "No user");
  requireActive(user);
  return doProcess(req);
}

private void requireActive(User u) {
  if (!u.isActive())
    throw new InactiveUserEx(u);
}
